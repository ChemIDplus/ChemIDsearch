The JSON results for substances include references to Types and Sources. The following URLs return the data for these:

  • https://chem.nlm.nih.gov/api/data/meta/types
  • https://chem.nlm.nih.gov/api/data/meta/sources

The JSON results match the following interfaces: 


interface Source {{ '{' }}
	id :string;
	/** name */
	n :string;
	/** description */
	d :string;
	/** elementTypeId */
	t :number;
	/** preUrl */
	u :string;
}
interface Type {{ '{' }}
	id :number;
	name :string;
	table :string;
}
